#Practical-2
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit
# Define the Gompertz function
def gompertz(x, a, b, c):
return a * np.exp(-b * np.exp(-c * x))
# Generate synthetic data (for demonstration)
np.random.seed(42)
x_data = np.linspace(0, 10, 100) # Time points
a_true, b_true, c_true = 100, 2, 0.5 # True parameters
y_data = gompertz(x_data, a_true, b_true, c_true) + np.random.normal(0, 2,
size=len(x_data))# Add noise
# Fit the Gompertz function to the data
popt, pcov = curve_fit(gompertz, x_data, y_data, p0=[100, 2, 0.5]) # Initial guess
# Extract fitted parameters
a_fit, b_fit, c_fit = popt
print(f"Fitted parameters: a={a_fit:.2f}, b={b_fit:.2f}, c={c_fit:.2f}")
# Plot the data and fitted curve
plt.scatter(x_data, y_data, label="Data", color='blue', alpha=0.6)
plt.plot(x_data, gompertz(x_data, *popt), label="Fitted Gompertz Curve", color='red')
plt.xlabel("Time")
plt.ylabel("Growth")
plt.legend()
plt.title("Gompertz Curve Fitting")
plt.show()

#Practical-3
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit
def logistic(x, L, k, x0):
return L/(1+np.exp(-k*(x-x0)))
#Generate synthetic data
np.random.seed(42)
x_data= np.linspace(0, 10, 100)
y_data= logistic(x_data, L=10, k=1, x0=5) + np.random.normal(0,0.5, size=len(x_data))
# Fit logistic curve to data
popt, _ = curve_fit(logistic, x_data, y_data, p0=[10, 1, 5])
L_fit, k_fit, x0_fit = popt
# Generate fitted curve
x_fit = np.linspace(0, 10, 100)
y_fit = logistic(x_fit, L_fit, k_fit, x0_fit)
# Plot results
plt.scatter(x_data, y_data, label='Data', alpha=0.6)
plt.plot(x_fit, y_fit, color='red', label=f'Fitted Logistic Curve\nL={L_fit:.2f},
k={k_fit:.2f}, x0={x0_fit:.2f}')
plt.xlabel('X')
plt.ylabel('Y')
plt.legend()
plt.title('Logistic Curve Fitting')
plt.show()

#Practical-5
import numpy as np
import pandas as pd
import statsmodels.api as sm
import matplotlib.pyplot as plt
#Sample time series data (e.g., monthly sales data for 3 years)
data={
"Month":pd.date_range(start="2020-01-01",periods=36,freq="M"),
"Sales":[200, 220, 250, 270, 300, 320, 340, 310, 280, 260, 240, 230,
210, 225, 255, 275, 310, 330, 345, 315, 285, 265, 245, 235,
215, 230, 260, 280, 320, 340, 350, 320, 290, 270, 250, 240]
}
df=pd.DataFrame(data)
df["Month_Num"]=np.arange(1,len(df)+1)
# Step 1: Fit a Linear Trend Model (Sales = a + b*Time)
X = sm.add_constant(df["Month_Num"]) # Add constant for intercept
model = sm.OLS(df["Sales"], X).fit()
df["Trend"] = model.predict(X)
# Step 2: Calculate the Ratio-to-Trend (Sales / Trend)
df["Ratio"] = df["Sales"] / df["Trend"]
# Step 3: Compute Average Seasonal Indices
df["Month_Only"] = df["Month"].dt.month # Extract month number
seasonal_indices = df.groupby("Month_Only")["Ratio"].mean()
# Normalize the indices to sum up to the number of periods (12 for monthly data)
seasonal_indices = seasonal_indices / seasonal_indices.mean()
# Display the seasonal indices
print("Seasonal Indices (Ratio-to-Trend Method):")
print(seasonal_indices)
# Step 4: Plot the Seasonal Indices
plt.figure(figsize=(10, 5))
plt.bar(seasonal_indices.index, seasonal_indices.values, color='skyblue')
plt.xlabel("Month")
plt.ylabel("Seasonal Index")
plt.title("Seasonal Indices (Ratio-to-Trend Method)")
plt.xticks(range(1, 13))
plt.show()


#Practical-6
import numpy as np
import pandas as pd
# Sample Data: Monthly Sales Data
data = {
'Month': pd.date_range(start='2020-01', periods=24, freq='M'), # 2 years of data
'Sales': [260, 280, 300, 320, 310, 330, 360, 380, 400, 420, 410, 430,
270, 290, 310, 330, 320, 340, 370, 390, 410, 430, 420, 440]
}
df = pd.DataFrame(data)
df['Month_Num'] = df['Month'].dt.month # Extract month number
# 1. Calculate Centered Moving Average (12-month moving average for monthly data)
df['CMA'] = df['Sales'].rolling(window=12, center=True).mean()
# 2. Compute Ratio-to-Moving Average (Sales / CMA)
df['RMA'] = df['Sales'] / df['CMA']
# 3. Compute Seasonal Indices (Average of RMA for each month)
seasonal_indices = df.groupby('Month_Num')['RMA'].mean()
# 4. Normalize the Seasonal Indices (so they sum to the number of seasons)
seasonal_indices = seasonal_indices / seasonal_indices.mean() * 12
# Display Seasonal Indices
print("Seasonal Indices:\n", seasonal_indices)

#Practical-7
import numpy as np
import pandas as pd
# Step 1: Input the quarterly sales data
data = {
'Year': [2022, 2023, 2024],
'Q1': [200, 210, 220],
'Q2': [250, 260, 270],
'Q3': [180, 190, 200],
'Q4': [220, 230, 240]
}
df = pd.DataFrame(data)
df
# Step 2: Compute Link Relatives
link_relatives = []
for i in range(len(df)):
row = []
if i > 0:
row.append((df.loc[i, 'Q1'] / df.loc[i - 1, 'Q4']) * 100) # Q1 vs Previous Q4
for j in range(1, 4):
row.append((df.iloc[i, j] / df.iloc[i, j - 1]) * 100) # Other quarters
link_relatives.append(row)
# Convert to DataFrame
link_rel_df = pd.DataFrame(link_relatives, columns=['Q1', 'Q2', 'Q3', 'Q4'])
# Step 3: Calculate Average Link Relatives
avg_link_relatives = link_rel_df.mean()
# Step 4: Adjust Seasonal Indices
sum_avg = avg_link_relatives.sum()
adjustment_factor = 400 / sum_avg
seasonal_indices = avg_link_relatives * adjustment_factor
#Display Results
print("\nAverage Link Relatives:")
print(avg_link_relatives)
print("\nSeasonal Indices:")
print(seasonal_indices)

import numpy as np
def variate_difference_variance(data):
"""
Calculate the variance of a random component using the variate difference method.
Parameters:
data (list or numpy array): A sequence of observations.
Returns:
float: Estimated variance.
"""
n = len(data)
if n < 2:
raise ValueError("Data must contain at least two observations.")
diff_sq = [(data[i+1] - data[i])**2 for i in range(n-1)]
variance = (1 / (2 * (n - 1))) * sum(diff_sq)
return variance
# Example usage
data = [10, 12, 15, 14, 18, 20, 19] # Example dataset
variance = variate_difference_variance(data)
print(f"Estimated variance: {variance:.4f}")

#Moving Average
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
np.random.seed(42)
data = np.cumsum(np.random.randn(100))
window = 5
moving_avg = pd.Series(data).rolling(window).mean()
plt.plot(data)
plt.plot(moving_avg)
plt.xlabel('Time')
plt.ylabel('Value')
plt.title('Moving Average')
plt.show()
#Expotential Smoothing
from statsmodels.tsa.holtwinters import ExponentialSmoothing
ses = ExponentialSmoothing(data).fit(smoothing_level = 0.2, optimized = False)
forecast = ses.forecast(20)
plt.figure(figsize = (12, 6))
plt.plot(data, label = 'Original Data')
plt.plot(ses.fittedvalues, label = 'Fitted Values')
# Plot the forecast starting from the end of the original data
plt.plot(np.arange(len(data), len(data) + len(forecast)), forecast, label = 'Forecast')
plt.legend()
plt.xlabel('Time')
plt.ylabel('Value')
plt.title('Exponential Smoothing')
plt.show()
#ARIMA
from statsmodels.tsa.arima.model import ARIMA
model = ARIMA(data, order = (2,1,2))
model_fit = model.fit()
forecast = model_fit.forecast(step = 10)
plt.plot(data, label = "Actual")
plt.plot(range(len(data), len(data) + 10), forecast, label = "Forecast")
plt.xlabel("Time")
plt.ylabel("Value")
plt.title("ARIMA Forecast")
plt.legend()
plt.show()
